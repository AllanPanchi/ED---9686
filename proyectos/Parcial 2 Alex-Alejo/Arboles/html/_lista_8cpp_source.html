324 int partition(std::vector<int>& arr, int low, int high) {
325 int pivot = arr[high];
326 int i = low - 1;
327
328 for (int j = low; j <= high - 1; j++) {
329 if (arr[j] <= pivot) {
330 i++;
331 std::swap(arr[i], arr[j]);
332 }
333 }
334 std::swap(arr[i + 1], arr[high]);
335 return i + 1;
336 }

338 // Implementación del algoritmo Quicksort
346 void quicksort(std::vector<int>& arr, int low, int high) {
347 if (low < high) {
348 int pi = partition(arr, low, high);
349 quicksort(arr, low, pi - 1);
350 quicksort(arr, pi + 1, high);
351 }
