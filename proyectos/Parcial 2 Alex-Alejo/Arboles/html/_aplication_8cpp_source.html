214// Función de partición para el Quicksort
215int partition(std::vector<float>& arr, int low, int high) {
216 float pivot = arr[high];
217 int i = low - 1;
218
219 for (int j = low; j <= high - 1; j++) {
220 if (arr[j] <= pivot) {
221 i++;
222 std::swap(arr[i], arr[j]);
223 }
224 }
225 std::swap(arr[i + 1], arr[high]);
226 return i + 1;
227}
228
235// Implementación del algoritmo Quicksort
236void quicksort(std::vector<float>& arr, int low, int high) {
237 if (low < high) {
238 int pi = partition(arr, low, high);
239 quicksort(arr, low, pi - 1);
240 quicksort(arr, pi + 1, high);
241 }
242}
